def letterNumber(fileName, letter): 
    file = open(fileName, "r") #1
    text = file.read() #2
    count = 0 #3
     
    for char in text: #4
          
        if char == letter: #5
            count += 1
      
    return count # 6
    
  

print(letterNumber ('moby-dick.txt', 'e')) # 7
